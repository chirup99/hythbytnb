I'll help you fetch profile details from the Delta Exchange API.## Fetch Profile Details from Delta Exchange API

Here's how to fetch complete profile details using the Delta Exchange API.

Endpoint: Get User Profile
Endpoint: GET /v2/profile

Description: Returns the complete user object with personal and account-related details.

Authentication: Required

Parameters: None

Profile Fields Returned
id (string): Unique user identifier
email (string): User's email address
account_name (string): Main account or subaccount name
first_name (string): User's first name
last_name (string): User's last name
dob (string): Date of birth (YYYY-MM-DD format)
country (string): User's country of residence
phone_number (string): User's phone number with country code
margin_mode (string): "isolated" or "portfolio"
pf_index_symbol (string): Portfolio index symbol (if in portfolio margin mode)
is_sub_account (boolean): Whether the account is a sub-account
is_kyc_done (boolean): Whether KYC verification is completed
Example Request

curl -X GET "https://api.india.delta.exchange/v2/profile" \
  -H "api-key: your_api_key" \
  -H "timestamp: 1542110948" \
  -H "signature: your_generated_signature" \
  -H "User-Agent: python-rest-client" \
  -H "Content-Type: application/json"

Response Schema
Success Response (200):

Properties:

success (boolean): Indicates if the request was successful (default: true)
result (object): User profile object containing all user details

import hashlib

base_url = 'https://api.india.delta.exchange'
api_key = 'your_api_key'
api_secret = 'your_api_secret'

def generate_signature(secret, message):
    """Generate HMAC SHA256 signature"""
    message = bytes(message, 'utf-8')
    secret = bytes(secret, 'utf-8')
    hash = hmac.new(secret, message, hashlib.sha256)
    return hash.hexdigest()

def fetch_profile():
    """Fetch user profile details"""
    
    # Prepare request
    method = 'GET'
    timestamp = str(int(time.time()))
    path = '/v2/profile'
    query_string = ''
    payload = ''
    
    # Generate signature
    signature_data = method + timestamp + path + query_string + payload
    signature = generate_signature(api_secret, signature_data)
    
    # Build headers
    headers = {
        'api-key': api_key,
        'timestamp': timestamp,
        'signature': signature,
        'User-Agent': 'python-rest-client',
        'Content-Type': 'application/json'
    }
    
    # Make request
    url = f'{base_url}{path}'
    
    try:
        response = requests.get(url, headers=headers, timeout=(3, 27))
        response.raise_for_status()
        data = response.json()
        
        if data.get('success'):
            profile = data['result']
            
            # Display profile details
            print("=" * 50)
            print("USER PROFILE DETAILS")
            print("=" * 50)
            print(f"User ID:           {profile['id']}")
            print(f"Account Name:      {profile['account_name']}")
            print(f"Full Name:         {profile['first_name']} {profile['last_name']}")
            print(f"Email:             {profile['email']}")
            print(f"Phone Number:      {profile['phone_number']}")
            print(f"Country:           {profile['country']}")
            print(f"Date of Birth:     {profile['dob']}")
            print(f"Margin Mode:       {profile['margin_mode']}")
            print(f"Portfolio Index:   {profile.get('pf_index_symbol', 'N/A')}")
            print(f"Is Sub-Account:    {profile['is_sub_account']}")
            print(f"KYC Status:        {'Completed' if profile['is_kyc_done'] else 'Pending'}")
            print("=" * 50)
            
            return profile
        else:
            print(f"Failed to fetch profile: {data}")
            return None
            
    except requests.exceptions.RequestException as e:
        print(f"Error fetching profile: {e}")
        if hasattr(e, 'response') and e.response is not None:
            print(f"Response: {e.response.text}")
        return None

# Fetch and display profile

    print(f"\nQuick Access - User ID: {user_id}, Username: {username}")

Error Handling
Always include proper error handling:


def fetch_profile_safe():
    """Fetch profile with comprehensive error handling"""
    try:
        method = 'GET'
        timestamp = str(int(time.time()))
        path = '/v2/profile'
        
        signature_data = method + timestamp + path
        signature = generate_signature(api_secret, signature_data)
        
        headers = {
            'api-key': api_key,
            'timestamp': timestamp,
            'signature': signature,
            'User-Agent': 'python-rest-client',
            'Content-Type': 'application/json'
        }
        
        response = requests.get(f'{base_url}{path}', headers=headers, timeout=(3, 27))
        
        # Check HTTP status
        if response.status_code == 401:
            print("Authentication failed. Check your API key and secret.")
            return None
        elif response.status_code == 403:
            print("Access forbidden. Check IP whitelist and API permissions.")
            return None
        elif response.status_code != 200:
            print(f"HTTP Error {response.status_code}: {response.text}")
            return None
        
        data = response.json()
        
        if not data.get('success'):
            print(f"API Error: {data.get('error', 'Unknown error')}")
            return None
        
        return data['result']
        
    except requests.exceptions.Timeout:
        print("Request timed out. Please try again.")
        return None
    except requests.exceptions.ConnectionError:
        print("Connection error. Check your internet connection.")
        return None
    except Exception as e:
        print(f"Unexpected error: {e}")
        return None

# Use the safe function

Summary
To fetch profile details from Delta Exchange:

Basic Profile: Use GET /v2/profile for user ID, name, email, and account settings
Trading Preferences: Use GET /v2/users/trading_preferences for VIP level, volume, and notification settings
Wallet Balances: Use GET /v2/wallet/balances for account balances
Sub-Accounts: Use GET /v2/sub_accounts to list all sub-accounts (from main account only)
All endpoints require proper authentication with API key and signature. The profile endpoint is the primary source for user identification and basic account information.
